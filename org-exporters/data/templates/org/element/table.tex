\@bind[opt]{%!
  \@this@bodies@head@head@list{%!
    \@match[this.alignment]{
      \@case[left]{l}
      \@case[right]{r}
      \@case[center]{c}
      \@default{l}
    }%!
  }%!
}%!
\@bind[thead]{%!
  \@if[this.head.nonempty]{%!
    \@this@head@list[as=row, with=lrow]\\\\
        \hline%!
  }%!
}%!
\@bind[tbodies]{\@this@bodies@list[as=body, with=lbody, intercalate={    \hline}]}%!
\@bind[lbody]{\@body@list[as=row, with=lrow, intercalate={\\\\}]}%!
\@bind[lrow]{
      \@row@list[as=cell, with=lcell, intercalate={ & }]%!
}%!
\@bind[lcell]{\@cell@content}%!

\@sep%!
\begin{center}
  \begin{tabular}{\@opt}%!
    \@thead%!
    \@tbodies
  \end{tabular}
\end{center}%!
\@sep%!
