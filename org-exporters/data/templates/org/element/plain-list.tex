\@bind[my.item-params]{%!
  \@match[item.checkbox]{
    \@case[true]{[{$\boxtimes$}]}
    \@case[false]{[{$\square$}]}
    \@case[partial]{[{$\boxminus$}]}
    \@default{%!
      \@if[item.descriptive-tag.nonempty]{%!
        [{\@item@descriptiveTag}]%!
      }%!
    }
  }%!
}%!

\@bind[my.list-env]{%!
  \@match[this.type]{
    \@case[unordered]{itemize}
    \@case[ordered-num]{enumerate}
    \@case[ordered-alpha]{enumerate}
    \@case[descriptive]{description}
  }%!
}%!

\@sep%!
\begin{\@my@listEnv}%!
  \@this@items@list[as=item]{
    \item\@my@itemParams \@item@content%!
  }
\end{\@my@listEnv}%!
\@sep%!
