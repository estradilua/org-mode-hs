\@bind[my:hlevel]{%!
  \@match[this.level]{
    \@case[1]{\section}
    \@case[2]{\subsection}
    \@case[3]{\subsubsection}
    \@case[4]{\paragraph}
    \@case[5]{\subparagraph}
    \@case[6]{\subsubparagraph}
  }%!
}%!

\@bind[my:headline]{%!
  \@if[section.todo]{{\bfseries\sffamily \@section@todo@name} }%!
  \@if[section.priority]{\framebox{\\#\@section@priority}}%!
  \@section@title%!
  \@if[section.tags.nonempty]{\hfill{} \@section@tags@list[intercalate={:}]{\@this}}%!
}%!

\@match[this.type]{
  \@case[normal]{
    \@this@list[as=section]{%!
      \@my@hlevel{}{\@my@headline}
      \label{sec:\@section@id}
      \@section@children
      \@section@subsections
    }%!
  }
  \@case[over-level]{
    \begin{enumerate}%!
      \@this@list[as=section]{
        \item \@my@headline
        \label{sec:\@section@id}

        \@section@children%!
        \@section@subsections%!
      }
    \end{enumerate}
  }
}%!
